  3. # 是否使用更直观的渲染方式
  4. fancy_render: true

  7. trim_transparency: true

  9. # 图片的最大尺寸,默认宽度为35格,高度为20格
  10. max_width: 35
  11. max_height: 20


  14. # 发送冷却设置 #

  16. # 需要多久时间才能再接收一次图片(单位:秒)
  17. cooldown: 3
  18. .
  19. # 启用的话直接发链接也会算入冷却时间
  20. strict_cooldown: false


  23. # 过滤设置 #

  25. bad_words:

  27.   # 如果图片带有攻击别人的语句/字则不会渲染图片,这个只适用于指令发送的消息.
  28.   enabled: false

  30.   # 如果图片包含攻击别人的语句/字则不会渲染图片,启用他以屏蔽链接发送的图片.
  31.   # 链接也发不出来
  32.   # 哪怕他没有发送图片的权限
  33.   remove_message: false

  35.   #要屏蔽的单词,用双引号包裹 中间用逗号分割 示例:
  36.   # exclusions: ["脏话","不利的话, "恶心的话"]
  37.   exclusions: []

  39. explicit_content:

  41.   #这个是填API密钥的
  42.   #请前往这个网站获取:[url]https://moderatecontent.com/signup[/url]
  43.   #只需要一个电子邮箱即可获取
  44.   #要查看你的服务器检测图片多少次了,请前往这个链接[url]https://client.moderatecontent.com/[/url]
  45.   #每月只可扫描一万多张图片，一般情况下是够用的
  46.   #这个密钥是用于屏蔽图片的,屏蔽让人不适的图片，会延长点渲染时间,而且也只是可选项
  47.   api_key: ""

  49.   #你需要把上面的API密钥填完才可启用此功能
  50.   #启用后如果图片带有令人不适的内容，将会屏蔽这个图片
  51.   #这会延长点渲染时间,这个只适用于指令发送的消息.
  52.   enabled: false

  54.   #同上,不过启用后会也屏蔽链接发送的消息
  55.   #链接也发不出来
  56.   #同时包括没有发送权限图片的玩家发的链接，只要是链接他就会检测
  57.   remove_message: false

  59. # 出现问题时发送报错到控制台,以方便查找问题原因
  60. debug: false

权限及指令

权限:
chatimage.use 用户发送图片的权限
chatimage.admin 管理员权限(默认op)
指令:
/ignoreimages(默认所有玩家都可使用)
开启/关闭接收图片
​
/chatimage reload(需要chatimage.admin权限)
重新加载配置文件

/chatimage send <玩家> <url> [smooth] [trim] [宽度] [高度] [文本](需要chatimage.use权限)
向特定玩家发送图片，后面如果的选项如果未填写将用默认的，文本那一栏至此换行服(/n)及颜色代码。
